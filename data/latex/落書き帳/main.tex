\scalebox{2}{\fbox{1}}　\;(20点)
\\[5mm]\indent
　数列$\{a_n\}$と関数$f(x)$を
\[f(x)=\log(1+x^2), \;a_n=\dfrac{\{(n^2+1^2)(n^2+2^2)\cdots(n^2+n^2)\}^{\frac{1}{n}}}{n^2}\]
　と定める．\\[5mm]
\begin{enumerate}
\item[(1)]　極限値，$\displaystyle\lim_{n\rightarrow\infty}a_n$を求めよ．また以下この値を$A$とする．
\item[(2)]　$0\leq{x}\leq{1}$の範囲において$f''(x)\geq{0}$を示せ．
\item[(3)]　$-1\leq{x}\leq{1}$の範囲で不等式，
\[1+x\leq{e^x}\leq{1+x+x^2}\]
を示せ．
\item[(4)]　$n$，$k$を自然数とする．等式，
\[\dfrac{f\bigg(\dfrac{k}{n}\bigg)+f \bigg(\dfrac{k-1}{n}\bigg)}{2n}-\displaystyle\int_{\frac{k-1}{n}}^{\frac{k}{n}}f(x)dx=-\dfrac{1}{2}\displaystyle\int_{\frac{k-1}{n}}^{\frac{k}{n}}\bigg(x-\dfrac{k}{n}\bigg)\bigg(x-\dfrac{k-1}{n}\bigg)f''(x)dx\]
を示し、これを用いて以下のiを示せ．
\item[]
\item[]　i.\;\;ある定数$C$が存在し，すべての自然数$n$に対して
\[\left\lvert\displaystyle\sum_{k=1}^n\dfrac{f \bigg(\dfrac{k}{n}\bigg)+f \bigg(\dfrac{k-1}{n}\bigg)}{2n}-\log{A}\right\rvert<\dfrac{C}{n^2}\]
　が成立する．
\item[]
\item[(4)]　極限値，$\displaystyle\lim_{n\rightarrow\infty}n(a_n-A)$を求めよ．
\end{enumerate}

\\[5mm]\indent
　微分可能な関数$f(x)$に対して命題$P$，$Q$を
\[P:\displaystyle\lim_{x\rightarrow{0}}\dfrac{f(\{f(x)\}^2)}{x}=a\,が成り立つ\]
\[Q:\displaystyle\lim_{x\rightarrow{0}}\dfrac{\log{(1+f(x^4))}}{x}=a\,が成り立つ\]





と定める．
\begin{enumerate}
\item[(1)]　命題，$P\Rightarrow{Q}$の真偽を求めよ．
\item[(2)]　命題，$Q\Rightarrow{P}$の真偽を求めよ．
\end{enumerate}
